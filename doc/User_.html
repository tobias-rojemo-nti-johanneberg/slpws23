def initialize(db, id)
  @db = db
  @id = id
  @data = @db.execute("SELECT username, perms, pfp FROM users WHERE id = ? LIMIT 1", @id)
  @name = @data[0]["username"]
  @perms = @data[0]["perms"]
  @pfp = @data[0]["pfp"]
end
